# Addition-Code-
Addition of two numbers


#include <iostream>
using namespace std;

int main() {
    int a, b, sum;

    
    cout << "Enter first number: ";
    cin >> a;

    
    cout << "Enter second number: ";
    cin >> b;

    // sum of the two numbers
    sum = a + b;

    // result
    cout << "The sum of the two numbers is " << sum << endl;

    return 0;
}
